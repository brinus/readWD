194void getPedestal(float *aWaveform, float &pedestal, float &stdv)
195{
196 pedestal = 0;
197 stdv = 0;
198 float vStdv1 = TMath::StdDev(100, aWaveform);
199 float vStdv2 = TMath::StdDev(100, aWaveform + 100);
200 if (vStdv1 < vStdv2)
201 {
202 pedestal = TMath::Mean(100, aWaveform);
203 stdv = vStdv1;
204 }
205 else
206 {
207 pedestal = TMath::Mean(100, aWaveform + 100);
208 stdv = vStdv2;
209 }
210}

222void getPedestal(TH1F *hSignal, float &pedestal, float &stdv)
223{
224 pedestal = stdv = 0;
225 int nBins = 100;
226 float meanSquares1 = 0;
227 float mean1 = 0;
228 float stdv1 = 0;
229 float meanSquares2 = 0;
230 float mean2 = 0;
231 float stdv2 = 0;
232 float x = 0;
233 for (int i = 0; i < nBins; ++i)
234 {
235 x = hSignal->GetBinContent(i + 1);
236 mean1 += x;
237 meanSquares1 += x * x;
238 x = hSignal->GetBinContent(i + nBins + 1);
239 mean2 += x;
240 meanSquares2 += x * x;
241 }
242 mean1 /= nBins;
243 mean2 /= nBins;
244 stdv1 = sqrt(meanSquares1 / nBins - mean1 * mean1);
245 stdv2 = sqrt(meanSquares2 / nBins - mean2 * mean2);
246 if (stdv1 < stdv2)
247 {
248 pedestal = mean1;
249 stdv = stdv1;
250 }
251 else
252 {
253 pedestal = mean2;
254 stdv = stdv2;
255 }
256}

269void subtractSineNoise(float *wfData, float *wfTime, float &ampl, float &phi, char *name = 0)
270{
271 float freq = gCONFIG.noiseFrequency;
272 TMatrixD M(3, 3);
273 TVectorD v(3);
274 for (int i = 0; i < 200; ++i)
275 {
276 float co = TMath::Cos(freq * wfTime[i]);
277 float si = TMath::Sin(freq * wfTime[i]);
278 M[0][0] += co * co;
279 M[0][1] += co * si;
280 M[0][2] += co;
281 M[1][1] += si * si;
282 M[1][2] += si;
283 M[2][2] += 1;
284 v[0] += co * wfData[i];
285 v[1] += si * wfData[i];
286 v[2] += wfData[i];
287 }
288 M[1][0] = M[0][1];
289 M[2][0] = M[0][2];
290 M[2][1] = M[1][2];
291 M.Invert();
292 v = M * v;
293 ampl = TMath::Sqrt(v[0] * v[0] + v[1] * v[1]);
294 phi = TMath::ATan2(v[0], v[1]);
295 if (name)
296 {
297 TGraph *aGraph = new TGraph(SAMPLES_PER_WAVEFORM, wfTime, wfData);
298 TF1 *noise = new TF1("SineNoise", "[0] * sin([1] * x + [2]) + [3]");
299 noise->SetNpx(1000);
300 noise->SetParameters(ampl, freq, phi, v[2]);
301 aGraph->GetListOfFunctions()->Add(noise);
302 aGraph->SetName(name);
303 aGraph->SetTitle(name);
304 aGraph->Write("", TObject::kOverwrite);
305 }
306 for (int i = 0; i < SAMPLES_PER_WAVEFORM; ++i)
307 {
308 wfData[i] -= ampl * TMath::Sin(freq * wfTime[i] + phi);
309 }
310}

320float getTimeStamp(EventHeader eh)
321{
322 static int lastDay = 0;
323 static int nDays = 0;
324 if (eh.serialNumber == 0)
325 {
326 lastDay = eh.day;
327 nDays = 0;
328 }
329 if (lastDay != eh.day)
330 {
331 lastDay = eh.day;
332 ++nDays;
333 }
334 float timestamp = eh.ms / 1000. + eh.sec + eh.min * 60 + eh.hour * 3600 + nDays * 86400;
335 return timestamp;
336}

349int ReadAnEvent(std::ifstream *file, EventHeader &eh, std::vector<float *> &wfData, std::vector<unsigned short> *tCell)
350{
351 // Input: file - std::ifstream opened with Initialise() and the current position at the beginning of the EHDR.
352 // eh - to this variable the event header will be read
353 // wfData - to this variable the waveform data will be read. Memory already properly alocated!
354 // tCell - if valid address, the trigger cells will be stored to this address.
355 // Return: 0 on success, 1 for eof, 2 otherwise
356 // the std::ifstream file will be at the very end of the event - next thing read should be the new EHDR.
357 // Further: The details of the read data depends on gCONFIG.nChannelsPerBoard.
358
359 char word[5]; // A default word to read from a file
360 word[4] = '\0'; // 4 bytes to read + 1 \0 character to terminate for printing purposes
361 unsigned short voltages[SAMPLES_PER_WAVEFORM]; // Voltages for a single channel to read from the file.
362 unsigned short index = 0; // Current channel number;
363
364 // Get the last position in the file
365 static long lastPos = 0;
366 static long eventSize = 0;
367 if (gCONFIG.firstOfRun || not lastPos || not eventSize)
368 {
369 long curPos = file->tellg();
370 file->seekg(0, file->end);
371 lastPos = file->tellg();
372 file->seekg(curPos);
373 if (gCONFIG.runMode == 1)
374 {
375 // WDB event encoding
376 eventSize += sizeof(EventHeader); // 1 Event header
377 eventSize += 4 * gCONFIG.nChannelsPerBoard.size(); // 4 byte board header per board
378 int nChannelsTot = 0;
379 for (int nCh : gCONFIG.nChannelsPerBoard)
380 nChannelsTot += nCh;
381
382 // 4 Byte channel header, 4 byte scaler, 4 byte Trigger header + data
383 eventSize += (12 + sizeof(voltages)) * nChannelsTot;
384 }
385 else
386 {
387 // DRS event encoding
388 eventSize += sizeof(EventHeader); // 1 Event header
389 eventSize += 8 * gCONFIG.nChannelsPerBoard.size(); // 4 byte board header per board + trigger header
390 int nChannelsTot = 0;
391 for (int nCh : gCONFIG.nChannelsPerBoard)
392 nChannelsTot += nCh;
393
394 // 4 Byte channel header, 4 byte scaler + data
395 eventSize += (8 + sizeof(voltages)) * nChannelsTot;
396 }
397 gCONFIG.firstOfRun = false;
398 }
399 DEBUG << "File status: " << file->tellg() << "/" << lastPos << " Event Size: " << eventSize << std::endl;
400
401 // Assert that the file is not at its end
402 if (lastPos - file->tellg() < eventSize)
403 {
404 DEBUG << "File is at its end: " << file->tellg() << "/" << lastPos << " Event Size: " << eventSize << std::endl;
405 return 1;
406 }
407
408 // Read event header first
409 file->read((char *)&eh, sizeof(eh));
410 if (std::memcmp(eh.tag, "EHDR", 4) != 0)
411 {
412 std::cerr << "!! no valid event header found. Found " << eh.tag << "instead" << std::endl;
413 return 2;
414 }
415 DEBUG << "Reading event " << eh.serialNumber << std::endl;
416
417 // Looping over all channels and all boards
418 for (unsigned int board = 0; board < gCONFIG.nChannelsPerBoard.size(); ++board)
419 {
420 file->read(word, 4);
421 if (std::memcmp(word, "B#", 2) != 0)
422 {
423 std::cerr << "!! No valid board header found. Found " << word << "instead" << std::endl;
424 return 2;
425 }
426 DEBUG << " -found data for board " << *(short *)(word + 2) << std::endl;
427 if (gCONFIG.runMode == 0)
428 {
429 // DRS encoding
430 // Read trigger cell
431 file->read(word, 4);
432 if (std::memcmp(word, "T#", 2) != 0)
433 {
434 std::cerr << "No valid trigger cell found. Found " << word << "instead" << std::endl;
435 return 2;
436 }
437 DEBUG << "Trigger cell: " << *(unsigned short *)(word + 2) << std::endl;
438 if (tCell)
439 {
440 for (unsigned int i = 0; i < tCell->size(); ++i)
441 tCell->at(i) = *(unsigned short *)(word + 2);
442 }
443 }
444 for (unsigned int ch = 0; ch < gCONFIG.nChannelsPerBoard.at(board); ++ch)
445 {
446 // Read channel header
447 file->read(word, 4);
448 if (word[0] != 'C')
449 {
450 std::cerr << "No valiad channel header found. Found " << word << "instead" << std::endl;
451 return 2;
452 }
453 DEBUG << "Found data for channel " << word << std::endl;
454
455 // Read scaler - and ignore it for now
456 file->read(word, 4);
457 if (gCONFIG.runMode == 1)
458 {
459
460 // Read trigger cell
461 file->read(word, 4);
462 if (std::memcmp(word, "T#", 2) != 0)
463 {
464 std::cerr << "No valid trigger cell found. Found " << word << "instead" << std::endl;
465 return 2;
466 }
467 DEBUG << "Trigger cell: " << *(unsigned short *)(word + 2) << std::endl;
468 if (tCell)
469 tCell->at(index) = *(unsigned short *)(word + 2);
470 }
471
472 // Read voltages from file
473 file->read((char *)voltages, sizeof voltages);
474
475 // Transfere the voltages to the wfData
476 for (int bin = 0; bin < SAMPLES_PER_WAVEFORM; ++bin)
477 {
478 wfData.at(index)[bin] = (voltages[bin] / 65536. + eh.rangeCenter / 1000. - 0.5);
479 }
480 ++index;
481 }
482 }
483 return 0;
484}

492std::ifstream *Initialise(std::string filename)
493{
494 // Input: path- / filename to the .dat file to be read.
495 // timeBinWidth: a std::vector to contain the time width of each bin
496 // Return: open std::ifstream at position of first event header in case of success
497 // Return: NULL-pointer in case of failure.
498 // Further initialise:
499 // - nChannelsPerBoard
500 // - timeBinWidth
501 // - integrationWindows (in case runMode >= 1)
502
503 unsigned int nBoards = 1;
504 unsigned int board = 0;
505 float *times = 0;
506 char word[5];
507 word[4] = '\0';
508 std::ifstream *file = new std::ifstream;
509 file->open(filename, std::ios::in | std::ios::binary);
510
511 // Reset the configurations
512 gCONFIG.firstOfRun = true;
513 gCONFIG.nChannelsPerBoard.resize(0);
514 gCONFIG.integrationWindows.resize(0);
515 gCONFIG.timeBinWidth.resize(0);
516
517 // Return if unable to open the file
518 if (!file->is_open())
519 {
520 std::cerr << "!! Could not open file " << filename << std::endl;
521 delete file;
522 return 0;
523 }
524 FileHeader fh;
525 file->read((char *)&fh, sizeof(fh));
526
527 // Check file header
528 if (fh.tag[0] != 'D' || fh.tag[1] != 'R' || fh.tag[2] != 'S')
529 {
530 std::cerr << "!! No suitable file header in file " << filename << std::endl;
531 file->close();
532 delete file;
533 return 0;
534 }
535 if (fh.version - '0' < 8)
536 {
537 // It is a DRS board
538 gCONFIG.runMode = 0;
539 std::cout << "DRS Version " << fh.version << " : DRS Board" << std::endl;
540 }
541 else
542 {
543 // It is a WDB
544 gCONFIG.runMode = 1;
545 std::cout << "DRS Version " << fh.version << " : WDB encoding" << std::endl;
546 }
547
548 // Skip the 'TIME' time header
549 file->seekg(4, file->cur);
550 file->read(word, 4);
551
552 if (word[0] != 'B')
553 {
554 std::cerr << "!! No board header found in file " << filename << std::endl;
555 file->close();
556 delete file;
557 return 0;
558 }
559 std::cout << "Board 1: " << word[0] << word[1] << *(short *)(word + 2) << std::endl;
560 gCONFIG.nChannelsPerBoard.push_back(0);
561 while (word[0] == 'C' || word[0] == 'B')
562 {
563 file->read(word, 4);
564 if (word[0] == 'C')
565 {
566 // It's a channel
567 std::cout << " -> " << word << std::endl;
568 gCONFIG.nChannelsPerBoard.at(board) += 1;
569 times = new float[SAMPLES_PER_WAVEFORM];
570 file->read((char *)times, sizeof(float) * SAMPLES_PER_WAVEFORM);
571 gCONFIG.timeBinWidth.push_back(times);
572 }
573 else if (word[0] == 'B')
574 {
575 ++board;
576 ++nBoards;
577 std::cout << "Board " << nBoards << ": " << word[0] << word[1] << *(short *)(word + 2) << std::endl;
578 }
579 }
580
581 // Now an event header should have been read
582 if (word[0] != 'E' || word[1] != 'H' || word[2] != 'D' || word[3] != 'R')
583 {
584 std::cerr << "!! Event header not found: Expected EHDR, got " << word << std::endl;
585 Debug << "Cleaning up ... " << std::endl;
586 file->close();
587 for (board = 0; board < gCONFIG.timeBinWidth.size(); ++board)
588 {
589 delete[] gCONFIG.timeBinWidth.at(board);
590 }
591 delete file;
592 return 0;
593 }
594
595 // The position in the std::ifstream should be at the beginning of the EHDR
596 file->seekg(-4, file->cur);
597 return file;
598}

606int GetIntegrationBounds(std::ifstream *file)
607{
608 // GetIntegrationBounds:
609 // Input: std::ifstream pointing to the .dat file at the beginning of the first EHDR
610 // Determines the integration windows (bins)
611 // Writes the Sample Signal Histos to the open root file
612
613 // Current position in the std::ifstream to restore later
614 int currentPos = file->tellg();
615
616 // Create SampleSignal histograms
617 int nChannelsTot = 0;
618 char name[32];
619 for (int i : gCONFIG.nChannelsPerBoard)
620 nChannelsTot += i;
621 TH1F **hSampleSig = new TH1F *[nChannelsTot];
622 for (int ch = 0; ch < nChannelsTot; ++ch)
623 {
624 sprintf(name, "hSampleSig%03d", ch);
625 hSampleSig[ch] = new TH1F(name, name, SAMPLES_PER_WAVEFORM, -0.5, SAMPLES_PER_WAVEFORM - 0.5);
626
627 // Attribute the histos with the root output file. Memory will be freed when closing the file
628 hSampleSig[ch]->SetDirectory(gCONFIG.theFile);
629 Debug << "Created histogram " << name << std::endl;
630 }
631
632 // Read some data and fill the histos
633 EventHeader eh;
634 std::vector<float *> wfData;
635 wfData.resize(nChannelsTot, 0);
636 for (int i = 0; i < nChannelsTot; ++i)
637 wfData.at(i) = new float[SAMPLES_PER_WAVEFORM];
638 float pedestal = 0;
639 float pedStdv = 0;
640 unsigned int nEvents = 0;
641 int retVal = 0;
642 while (nEvents < gCONFIG.nSampleEvents && !file->eof())
643 {
644 retVal = ReadAnEvent(file, eh, wfData, 0);
645 if (retVal)
646 break;
647 for (int ch = 0; ch < nChannelsTot; ++ch)
648 {
649 getPedestal(wfData.at(ch), pedestal, pedStdv);
650 for (int i = 0; i < SAMPLES_PER_WAVEFORM; ++i)
651 {
652 hSampleSig[ch]->Fill(i, wfData.at(ch)[i] - pedestal);
653 }
654 }
655 }
656 if (retVal < 2)
657 {
658 float peak, stdv, tau;
659 int peakBin;
660 TH1F *hSample = 0;
661 TF1 *gaus = new TF1("gaus", "gaus(0)");
662 TF1 *decay = new TF1("decay", "[0] * exp( -(x - [1]) / [2])");
663 TF1 *intWin = new TF1("intWindow", "[0]");
664 intWin->SetLineColor(419);
665 intWin->SetLineWidth(3);
666 IntegrationWindow iw;
667 std::cout << "Integration Windows:\nCh\tstart\tstop\n";
668
669 // Fit the samples if reasonable
670 for (int ch = 0; ch < nChannelsTot; ++ch)
671 {
672 hSample = hSampleSig[ch];
673 hSample->GetXaxis()->SetRange(2, 0.9 * SAMPLES_PER_WAVEFORM);
674 peak = hSample->GetMinimum();
675 peakBin = hSample->GetMinimumBin();
676 hSample->GetXaxis()->SetRange(1, SAMPLES_PER_WAVEFORM);
677 getPedestal(hSample, pedestal, pedStdv);
678 Debug << "peak : " << peak << " bin: " << peakBin << std::endl;
679 if (peak < pedestal - 5 * pedStdv)
680 {
681 // There is at least a five sigma excess
682 // This channel probably collected a waveform
683 decay->FixParameter(0, peak);
684 decay->FixParameter(1, peakBin);
685 decay->SetParameter(2, 10);
686 decay->SetRange(peakBin, 1000);
687 hSample->Fit(decay, "RQ0");
688 hSample->GetFunction("decay")->ResetBit(TF1::kNotDraw);
689 gaus->FixParameter(0, peak);
690 gaus->FixParameter(1, peakBin);
691 gaus->SetParameter(2, 10);
692 gaus->SetRange(1, peakBin);
693 hSample->Fit(gaus, "RQ0+");
694 hSample->GetFunction("gaus")->ResetBit(TF1::kNotDraw);
695 stdv = fabs(gaus->GetParameter(2));
696 tau = fabs(decay->GetParameter(2));
697 iw.start = peakBin - gCONFIG.intRise * stdv;
698 iw.stop = peakBin + gCONFIG.intDecay * tau;
699 if (iw.start < 0)
700 iw.start = 0;
701 if (iw.stop > SAMPLES_PER_WAVEFORM - 1)
702 iw.stop = SAMPLES_PER_WAVEFORM - 1;
703 }
704 else
705 {
706 // There is probably no signal - integrate over the whole time window
707 iw.start = 0;
708 iw.stop = SAMPLES_PER_WAVEFORM - 1;
709 }
710 intWin->SetRange(iw.start, iw.stop);
711 intWin->FixParameter(0, pedestal);
712 hSample->Fit(intWin, "RQ0+");
713 hSample->GetFunction("intWindow")->ResetBit(TF1::kNotDraw);
714 gCONFIG.integrationWindows.push_back(iw);
715 std::printf("%d\t%d\t%d\n", ch, iw.start, iw.stop);
716 }
717
718 // Deleting the functions
719 gaus->Delete();
720 decay->Delete();
721 intWin->Delete();
722 }
723
724 // Restore position in std::ifstream
725 file->seekg(currentPos);
726
727 // Clean up
728 for (float *f : wfData)
729 delete[] f;
730 for (int i = 0; i < nChannelsTot; ++i)
731 hSampleSig[i]->Write("", TObject::kOverwrite);
732
733 // Deleting the pointers to the histos, not the histos themselves
734 delete[] hSampleSig;
735
736 // Return 1 if eof was reached, 0 when nSampleEvents were read
737 return (nEvents < gCONFIG.nSampleEvents);
738}

749int ReadFile(std::ifstream *file)
750{
751 int retVal = 0;
752
753 // Initialise the tree
754 // 1. Number of channels and leaflist description
755 int nChannelsTot = 0;
756 std::string leaflistDescription = "ch0";
757 for (int nChannels : gCONFIG.nChannelsPerBoard)
758 {
759 nChannelsTot += nChannels;
760 }
761 for (int ch = 1; ch < nChannelsTot; ++ch)
762 {
763 leaflistDescription += ":ch";
764 leaflistDescription += std::to_string(ch);
765 }
766 Debug << leaflistDescription << std::endl;
767
768 // 2. Variables to fill the tree from
769 int fRun = gCONFIG.run;
770 int fEvent = 0;
771 float fTimestamp = 0;
772 float *fTime = new float[nChannelsTot];
773 float *fTimeLE = new float[nChannelsTot];
774 float *fAmplitude = new float[nChannelsTot];
775 float *fArea = new float[nChannelsTot];
776 float *fArea2 = new float[nChannelsTot];
777 float *fPed = new float[nChannelsTot];
778 float *fStdv = new float[nChannelsTot];
779 float *fThreshold = new float[nChannelsTot];
780 float *fStdvRaw = new float[nChannelsTot];
781 float *fSineAmplitude = new float[nChannelsTot];
782 float *fSinePhase = new float[nChannelsTot];
783
784 // TEST 001: New branch for trigger rate. Implement this by
785 // evaluating something similar to nEvents/elapsedTime.
786 float fTRate = 0; // TRate == T(rigger) Rate
787
788 // 3. TTree itself
789 TTree *theTree = new TTree("T", "WaveDREAM Tree");
790 theTree->SetDirectory(gCONFIG.theFile);
791 theTree->Branch("runnumber", &fRun);
792 theTree->Branch("event", &fEvent);
793 theTree->Branch("timestamp", &fTimestamp);
794 theTree->Branch("time", fTime, leaflistDescription.c_str());
795 theTree->Branch("timeLE", fTimeLE, leaflistDescription.c_str());
796 theTree->Branch("amplitude", fAmplitude, leaflistDescription.c_str());
797 theTree->Branch("area", fArea, leaflistDescription.c_str());
798 theTree->Branch("area2", fArea2, leaflistDescription.c_str());
799 theTree->Branch("ped", fPed, leaflistDescription.c_str());
800 theTree->Branch("Stdv", fStdv, leaflistDescription.c_str());
801 theTree->Branch("Threshold", fThreshold, leaflistDescription.c_str());
802
803 // TEST 001 ------------------------------
804 theTree->Branch("TRate", &fTRate);
805
806 if (gCONFIG.subtractSine)
807 {
808 theTree->Branch("StdvRaw", fStdvRaw, leaflistDescription.c_str());
809 theTree->Branch("SineAmplitude", fSineAmplitude, leaflistDescription.c_str());
810 theTree->Branch("SinePhase", fSinePhase, leaflistDescription.c_str());
811 }
812 Debug << "TTree initialised " << std::endl;
813 if (gCONFIG.debug)
814 theTree->Print();
815
816 // Start the mainloop
817 Debug << "Mainloop over the file started " << std::endl;
818 EventHeader eh;
819 std::vector<float *> wfData;
820 std::vector<float *> wfTime;
821 wfData.resize(nChannelsTot, 0);
822 wfTime.resize(nChannelsTot, 0);
823 for (int i = 0; i < nChannelsTot; ++i)
824 {
825 wfData.at(i) = new float[SAMPLES_PER_WAVEFORM];
826 wfTime.at(i) = new float[SAMPLES_PER_WAVEFORM];
827 }
828 std::vector<unsigned short> tCell;
829 tCell.resize(nChannelsTot, 0);
830 float t1, t2, dt;
831 int index = 0;
832
833 // Variables to be later filled into the tree
834 float timeCF, timeLE, ampl, area, area2, ped, stdv, stdvRaw, sinAmpl, phi, thr;
835 bool isSignal = false;
836 int minBin = 0;
837 float firstMin = 0;
838
839 // TEST 001 ------------------------------
840 float initTime = 0;
841 float prevTime = 0;
842
843 while (not file->eof())
844 {
845 retVal = ReadAnEvent(file, eh, wfData, &tCell);
846 if (retVal)
847 break;
848 if (eh.serialNumber % 100 == 0)
849 {
850 std::cout << "Processing Event " << eh.serialNumber << std::endl;
851 };
852
853 // TEST 001 ------------------------------
854 initTime = getTimeStamp(eh);
855 std::cout << initTime << std::endl;
856
857 // Calculate times for each channel
858 for (int ch = 0; ch < nChannelsTot; ++ch)
859 {
860 int bin = tCell.at(ch);
861 double time = 0;
862 for (int i = 0; i < SAMPLES_PER_WAVEFORM; ++i)
863 {
864 wfTime.at(ch)[i] = time;
865 time += gCONFIG.timeBinWidth.at(ch)[bin];
866 ++bin;
867 bin %= 1024;
868 }
869 }
870
871 // Align cell #0 of all channels
872 index = 0;
873 for (int nCh : gCONFIG.nChannelsPerBoard)
874 {
875 t1 = wfTime.at(index)[(1024 - tCell[index]) % 1024];
876 ++index;
877 for (int ch = 1; ch < nCh; ++ch)
878 {
879 t2 = wfTime.at(index)[(1024 - tCell[index]) % 1024];
880 dt = t1 - t2;
881 for (int i = 0; i < SAMPLES_PER_WAVEFORM; ++i)
882 {
883 wfTime.at(index)[i] += dt;
884 }
885 ++index;
886 }
887 }
888
889 // Analysis of individual channels
890 for (int ch = 0; ch < nChannelsTot; ++ch)
891 {
892 float *aWF = wfData.at(ch);
893
894 // Flip waveform if needed -> want a negative one
895 if (gCONFIG.sigWF == 1)
896 {
897 for (int i = 0; i < SAMPLES_PER_WAVEFORM; ++i)
898 aWF[i] *= -1;
899 Debug << "Positive Waveform - flipping" << std::endl;
900 }
901
902 // Subtract sine noise if activated
903 if (gCONFIG.subtractSine)
904 {
905 getPedestal(aWF, ped, stdvRaw);
906 if (eh.serialNumber < gCONFIG.nSaveEvents)
907 {
908 char name[32];
909 sprintf(name, "event%03d_ch%03d_raw", eh.serialNumber, ch);
910 subtractSineNoise(aWF, wfTime[ch], sinAmpl, phi, name);
911 }
912 else
913 {
914 subtractSineNoise(aWF, wfTime[ch], sinAmpl, phi);
915 }
916 }
917
918 // Get the pedestal
919 getPedestal(aWF, ped, stdv);
920
921 // Check for a signal waveform - aka 5 sigma excess
922 // May needs a somewhat less restrictive cut
923 isSignal = false;
924 IntegrationWindow iw = gCONFIG.integrationWindows.at(ch);
925 for (int i = iw.start; i < iw.stop && not isSignal; ++i)
926 {
927 if ((ped - aWF[i]) > 5 * stdv)
928 isSignal = true;
929 }
930
931 // Fill a graph if needed:
932 if (eh.serialNumber < gCONFIG.nSaveEvents)
933 {
934 TGraph *aGraph = new TGraph(SAMPLES_PER_WAVEFORM, wfTime[ch], aWF);
935 TF1 *pedestal = new TF1("pedestal", "[0]", wfTime[ch][iw.start], wfTime[ch][iw.stop]);
936 pedestal->SetParameter(0, ped);
937 aGraph->GetListOfFunctions()->Add(pedestal);
938 TString name = TString::Format("event%03d_ch%03d", eh.serialNumber, ch);
939 aGraph->SetName(name);
940 aGraph->SetTitle(name);
941 aGraph->Write("", TObject::kOverwrite);
942 }
943 if (isSignal)
944 {
945 // Find peak
946 ampl = firstMin = 0;
947 for (int bin = iw.start + 1; bin < iw.stop - 2; ++bin)
948 {
949 if (ampl > aWF[bin] - ped)
950 {
951 // The maximal amplitude in the intergation window
952 ampl = aWF[bin] - ped;
953 }
954 if (firstMin > -gCONFIG.leThreshold && aWF[bin] < aWF[bin - 1] && aWF[bin] < aWF[bin + 1] && aWF[bin] < aWF[bin + 2])
955 {
956 firstMin = aWF[bin] - ped;
957 minBin = bin;
958 }
959 }
960
961 // CF + LE time extraction
962 // start at minBin where the first (local) min has been found.
963 // Then move to earlier times to find the closest crossing.
964 timeLE = -1e+5;
965 timeCF = -1e+5;
966 for (int bin = minBin; bin > iw.start; --bin)
967 {
968 // DEBUG << aWF[bin] - ped << " " << -gCONFIG.leThreshold << " " << gCONFIG.cfFraction * firstMin << std::endl;
969 // LE
970 if (timeLE < 0 && aWF[bin] - ped > -gCONFIG.leThreshold)
971 {
972 timeLE = (-gCONFIG.leThreshold - aWF[bin] + ped) / (aWF[bin + 1] - aWF[bin]) * (wfTime[ch][bin + 1] - wfTime[ch][bin]) + wfTime[ch][bin];
973 }
974
975 // Constant Fraction
976 if (timeCF < 0 && aWF[bin] - ped > gCONFIG.cfFraction * firstMin)
977 {
978 timeCF = (gCONFIG.cfFraction * firstMin - aWF[bin] + ped) / (aWF[bin + 1] - aWF[bin]) * (wfTime[ch][bin + 1] - wfTime[ch][bin]) + wfTime[ch][bin];
979 }
980 if (timeCF > 0 && timeLE > 0)
981 break;
982 }
983
984 // Charge Integration
985 area = 0;
986 area2 = 0;
987 for (int bin = iw.start; bin < iw.stop; ++bin)
988 {
989 area += ped - aWF[bin];
990 area2 += (ped - aWF[bin]) * gCONFIG.timeBinWidth[ch][(tCell[ch] + bin) % 1024];
991 }
992 }
993 else
994 {
995 timeCF = -1e5;
996 timeLE = -1e5;
997 ampl = 0;
998 area = 0;
999 area2 = 0;
1000 thr = 0;
1001 }
1002 DEBUG << ch << "/" << nChannelsTot << std::endl;
1003 fEvent = eh.serialNumber;
1004 fTimestamp = getTimeStamp(eh);
1005 fTime[ch] = timeCF;
1006 fTimeLE[ch] = timeLE;
1007 fAmplitude[ch] = -ampl; // prefer a positive amplitude in the end;
1008 fArea[ch] = area;
1009 fArea2[ch] = area2;
1010 fPed[ch] = ped;
1011 fStdv[ch] = stdv;
1012 fThreshold[ch] = thr;
1013 fStdvRaw[ch] = stdvRaw;
1014 fSineAmplitude[ch] = sinAmpl;
1015 fSinePhase[ch] = phi;
1016 }
1017
1018 // TEST 001 ------------------------------
1019 if (prevTime != initTime)
1020 {
1021 fTRate = 1 / (initTime - prevTime);
1022 }
1023 else
1024 {
1025 fTRate = 0;
1026 }
1027
1028 theTree->Fill();
1029
1030 // TEST 001 ------------------------------
1031 prevTime = initTime;
1032 }
1033
1034 // Clean up at the end of the file
1035 Debug << "Finished reading - clean up " << std::endl;
1036 theTree->Write("", TObject::kOverwrite);
1037 delete[] fTime;
1038 delete[] fTimeLE;
1039 delete[] fAmplitude;
1040 delete[] fArea;
1041 delete[] fArea2;
1042 delete[] fPed;
1043 delete[] fStdv;
1044 delete[] fThreshold;
1045 delete[] fStdvRaw;
1046 delete[] fSineAmplitude;
1047 delete[] fSinePhase;
1048 for (float *f : wfData)
1049 delete[] f;
1050 return 0;
1051}
